@public
collection User {
  id: string; // wallet address

  publicKey: PublicKey;
  name: string; // wallet address
  token?: string;

  constructor (id: string) {
    this.id = id;
    this.name = id;
    this.publicKey = ctx.publicKey;
  }

  updateToken(token: string) {
    this.token = token;
  }
}
