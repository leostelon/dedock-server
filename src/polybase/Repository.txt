@public
collection Repository {
  id: string;
  name: string;
  tag: string;
  creator: string;
  image: string;
  timestamp: number;
  private: boolean;

  @index(timestamp);
  
  constructor(name:string, tag:string, image: string, creator:string) {
    this.id = name + ":" + tag;
    this.name = name;
    this.tag = tag;
    this.image = image;
    this.creator = creator;
    this.timestamp = Date.now();
    this.private = false;
  }

  makePrivate() {
    this.private = true;
  }

  makePublic() {
    this.private = false;
  }
}
